--------------------
DB Design
--------------------

users {
    first_name
    last_name
    email
    phone_number
    password
    avatar
    user_level
}

courses {
    title
    slug
    description
    thumbnail
    visibility
}

course_specializations {
    title
    ordering
}

course_course_specialization {
    course_id
    course_specialization_id
}

lessons {
    title
    ordering

    course_id (fk)
}

topics {
    title
    ordering

    lesson_id (fk)
}

content {
    title
    content
    ordering

    topic_id (fk)
}

enrollments {
    user_id
    courses (array of courses ids)
}
